
	;KEY 1
	CMP AL, 0F3H
	JE K1

	;KEY 2 and KEY 6 (keypad 2)
	CMP AL, 0F5H
	JE K2

	;KEY 3
	CMP AL, 0F6H
	JE K3

	;KEY 4
	CMP AL, 0FBH
	JE K4

	;KEY 5 & 7
	CMP AL, 0F9H
	JE K5

	;KEY 6 & *
	CMP AL, 0FAH
	JE K6

	;KEY 8
	CMP AL, 0FDH
	JE K8

	;KEY 9 & 0
	CMP AL, 0FCH
	JE K9

	;--------keypad 1 end-----------

	;--------keypad 2---------------
	
	;KEY 1
	CMP AL, 0D7H
	JE K1

	;KEY 2
	CMP AL, 0E7H
	JE K2

	;KEY 3
	CMP AL, 0F7H
	JE K3

	;KEY 4
	CMP AL, 0DBH
	JE K4

	;KEY 5
	CMP AL, 0EBH
	JE K5

	;KEY 6 
	CMP AL, 0F3H
	JE K6

	;KEY 7
	CMP AL, 0DDH
	JE K7

	;KEY 8
	CMP AL, 0EDH
	JE K8

	;KEY 9
	CMP AL, 0F5H
	JE K9

	;KEY 0
	CMP AL, 0EEH
	JE K0
	
	;JNE KEYPAD_EXIT

K1:
	XOR BX, BX
	MOV AL, KEYPAD_ROW
	CMP AL, 0FBH
	JE K6
	MOV BL, 01H
	MOV AL, SOUND_GREETING
	CALL VOICE_ON_QUEUE
	JMP ADD_BARCODE

K2:
	XOR BX, BX
	MOV AL, KEYPAD_ROW
	CMP AL, 0FDH
	JE K9
	MOV BL, 02H
	JMP ADD_BARCODE

K3:
	XOR BX, BX
	MOV BL, 03H
	JMP ADD_BARCODE
	
K4:
	XOR BX, BX
	MOV BL, 04H
	JMP ADD_BARCODE
K5:
	XOR BX, BX
	MOV AL, KEYPAD_ROW
	CMP AL, 0FDH
	JE K7
	MOV BL, 05H
	JMP ADD_BARCODE

K6:
	XOR BX, BX
	MOV AL, KEYPAD_ROW
	CMP AL, 0FEH
	JE K_STAR
	MOV BL, 06H
	JMP ADD_BARCODE
K7:
	XOR BX, BX
	MOV BL, 07H
	JMP ADD_BARCODE

K8:
	XOR BX, BX
	MOV BL, 08H
	JMP ADD_BARCODE

K9:
	XOR BX, BX
	MOV AL, KEYPAD_ROW
	CMP AL, 0FEH
	JE K0
	MOV BL, 09H
	JMP ADD_BARCODE

K0: 
	XOR BX, BX
	MOV BL, 00H
	JMP ADD_BARCODE

K_STAR:
	XOR BX, BX
	MOV CX, 08H
	CLEAR_BARCODE:
		MOV BX, CX
		DEC BX
		MOV BARCODE[BX], 00H
	LOOP CLEAR_BARCODE
	MOV BARCODE_I, 00H
	JMP KEYPAD_EXIT
	
K_HASH:
	XOR BX, BX
	MOV BL, 0AH
	JMP ADD_BARCODE

ADD_BARCODE:
	MOV AL, NUMBERS[BX]
	MOV BL, BARCODE_I
	MOV BARCODE[BX], AL
	INC BARCODE_I
	JMP KEYPAD_EXIT
